import datetime
def convert_strtime_minutes(time_str):
    if(time_str=='Less than 1 minute'):
        return 0
    time_str = time_str.replace(' hrs and','h').replace(' mins','m').replace(' min','m').replace(' hrs','h').replace(' hr','h')
    h,m = 0,0
    for item in time_str.split(' '):
        if 'h' in item:
            h = int(item.replace('h',''))
        if 'm' in item:
            m = int(item.replace('m',''))
    return (h*60)+m

import numpy as np
def format_rating_text(rating_text):
    if rating_text=='Not rated yet':
        return [np.NaN, np.NaN, 0]
    else:
        split_text = rating_text.split(' stars')
        stars = split_text[0].split(' ')[0]
        out_of = split_text[0].split(' out of ')[-1]
        number_of_ratings = split_text[-1].split(' ')[0]

        return [float(stars),float(out_of), int(number_of_ratings)]
